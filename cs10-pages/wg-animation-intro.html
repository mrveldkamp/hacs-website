Animation Basics

To create an animation, use the requestAnimatonFrame() function to execute code that will:

  • update elements on the canvas,
  • redraw the canvas,
  • and request another animation frame.

      let x = 100;

      requestAnimationFrame(draw);
      function draw() {
          // UPDATE ELEMENTS ON CANVAS
          x++;
      
          // DRAW CANVAS
          ctx.fillStyle = "white"
          ctx.fillRect(0, 0, cnv.width, cnv.height); // Background
      
          ctx.fillStyle = "green";
          ctx.fillRect(x, 100, 50, 50); // Rectangle
      
          // REQUEST ANIMATION FRAME
          requestAnimationFrame(draw);
      }

The requestAnimatonFrame() function enables the browser to determine the optimal FPS (frames per second) for the animation: usually around 60 frames per second.

Declaring, initializing, updating and using variables is critical to creating an animation.

Run Animation Till a Certain Value

An if statement may be used to control when to update animation variables. It is possible to only run an animation while a variable is within a certain range.


      let x = 100;

      requestAnimationFrame(draw);
      function draw() {
          // UPDATE ELEMENTS ON CANVAS
      
          // Animate if x is less than 400
          if (x < 400) {
              x++;
          }
      
          // DRAW CANVAS
          ctx.fillStyle = "white"
          ctx.fillRect(0, 0, cnv.width, cnv.height); // Background
      
          ctx.fillStyle = "green";
          ctx.fillRect(x, 100, 50, 50); // Rectangle
      
          // REQUEST ANIMATION FRAME
          requestAnimationFrame(draw);
      }

Run Animation For a Time

A variable that counts the number of frames (ie. frameCount) may be used to control when an animation should run.


      let x = 100;
      let frameCount = 0;
      
      requestAnimationFrame(draw);
      function draw() {
          // UPDATE ELEMENTS ON CANVAS
          frameCount++;
      
          // Animate if frameCount is less than 120
          // FPS is usually around 60 (so 120 frames == 2 seconds)
          if (frameCount < 120) {
              x++;
          }
      
          // DRAW CANVAS
          ctx.fillStyle = "white"
          ctx.fillRect(0, 0, cnv.width, cnv.height); // Background
      
          ctx.fillStyle = "green";
          ctx.fillRect(x, 100, 50, 50); // Rectangle
      
          // REQUEST ANIMATION FRAME
          requestAnimationFrame(draw);
      }

Reset Animation

An if statement may also be used to test the state of an animation and then modify any animation variables.


      let x = 100;

      requestAnimationFrame(animate);
      function animate() {
          // UPDATE ELEMENTS ON CANVAS
          x++;
      
          // Reset animation if x > 300
          if (x > 300) {
              x = 100;
          }
      
          // DRAW CANVAS
          ctx.fillStyle = "white"
          ctx.fillRect(0, 0, cnv.width, cnv.height); // Background
      
          ctx.fillStyle = "green";
          ctx.fillRect(x, 100, 50, 50); // Rectangle
      
          // REQUEST ANIMATION FRAME
          requestAnimationFrame(animate);
      }
