Create an Array with a Loop

Loops may be used to create arrays one item at a time. The push() method is used to add an element to the end of the array.


    // Create an array with identical values (100 exclamation marks)
    let identicalArray = [];
    for (let n = 0; n < 100; n++) {
        identicalArray.push("!");
    }

    // Create a random array (500 random decimals b/t 0 and 1)
    let randomArray = [];
    for (let n = 0; n < 500; n++) {
        randomArray.push(Math.random());
    }

    // Create an array from user input (5 colors)
    let inputArray = [];
    for (let n = 0; n < 5; n++) {
        inputArray.push(prompt("Please enter a color:"));
    }

    // Create a patterned array (even #'s b/t 0 and 50, inclusive)
    let patternArray = [];
    for (let num = 0; num <= 50; num += 2) {
        patternArray.push(num);
    }

Create an Array From a String

A string may be turned into an array with the split() method. The split() method takes in an argument that will be used as the token to separate the string.


    // Create arrays from strings with split()
    let namesString = "adam,ben,caleb,david";
    let namesArray = namesString.split(",");
    // -> ["adam", "ben", "caleb", "david"]

    let scoresString = "80;90;75;88;95;99;85";
    let scoresArray = scoresString.split(";");
    // -> ["80", "90", "75", "88", "95", "99", "85"]

    let sentence = "I love arrays";
    let wordArray = sentence.split(" ");
    // -> ["I", "love", "arrays"]
    let characterArray = sentence.split("");
    // -> ["I"," ","l","o","v","e"," ","a","r","r","a","y","s"]

The join() method may be used to turn an array back into a string. The join() method takes in an argument that will be used as the token to place between array elements.


    // Convert an array into a string with join()
    let wordArray = ["I", "love", "arrays"];
    let sentence = wordArray.join(" ");
    // -> "I love arrays"

Create an Array From a File

The fetch() and then() methods may be used to:

  • retrieve raw data from a file,
  • convert raw data to a string,
  • and then split string into an array.

    // Global Variable to store file contents
    let colorsArray;

    // colors.txt contains one color name per line
    fetch("colors.txt").then(convertData).then(processData);

    function convertData(rawData) {
      return rawData.text(); // convert file contents to string
    }

    function processData(data) {
      colorsArray = data.split("\r\n"); // Split string by end of line
    }

Same example as above but with Anonymous Arrow Functions (more concise)


    let colorsArray;

    fetch("colors.txt")
      .then((rawData) => rawData.text())
      .then((data) => colorsArray = data.split("\r\n"));

Create an Array of HTML Elements

HTML elements may be selected using querySelectorAll(). This method will take any valid css selector and all elements that match the css selector will be returned in a NodeList (similar to an array).


    // Select all paragraph elements with the class "special"
    let specialParaEls = document.querySelectorAll("p.special");

    // Add an event listener to all paragraph elements with the class special
    for (let i = 0; i < specialParaEls.length; i++) {
      specialParaEls[i].addEventListener("click", doSomethingSpecial);
    }
