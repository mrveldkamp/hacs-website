Local & Global Variables

Local Variable:

a variable declared inside of a function (only visible inside that function).

Global Variable:

a variable declared outside any functions (visible everywhere).


      let name = 'Megan'; // Global Variable

      function displayGreeting() {
        let timeOfDay = 'Morning'; // Local Variable

        console.log('Hello ' + name + '.'); // OK, global variable visible everywhere
        console.log('Have a good ' + timeOfDay + "!"); // OK, local variable visible in function
      }

      displayGreeting();
      console.log('Goodbye ' + name + '.'); // OK, global variable visible everywhere
      console.log('See you next ' + timeOfDay + '.'); // ERROR, timeOfDay is local to function
